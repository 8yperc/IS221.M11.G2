SELECT *

WHERE EXTRACT(MONTH FROM NGAYSINH) BETWEEN 6 AND 12

----2
SELECT MAKHOA
FROM dhcntt.SINHVIEN@dhcntt JOIN dhcntt.HOCPHI@dhcntt ON SINHVIEN.MASV = HOCPHI.MASV
WHERE GIOITINH = '1' AND HOCPHI.MASV IN 
(
	SELECT MASV
        FROM dhcntt.HOCPHI@dhcntt
	WHERE HOCKY = '1'
)

----3
CREATE TABLE temp AS
    SELECT MAKHOA, COUNT(MASV) AS SOLUONGSV
    FROM dhcntt.SINHVIEN@dhcntt
    GROUP BY MAKHOA
    ORDER BY SOLUONGSV DESC;
SELECT * FROM temp WHERE ROWNUM = 1

SELECT MAKHOA, COUNT(SINHVIEN.MASV) AS SOLUONGSV, SUM(SOTIEN) AS TOTAL
FROM dhcntt.SINHVIEN@dhcntt JOIN dhcntt.HOCPHI@dhcntt ON SINHVIEN.MASV = HOCPHI.MASV
GROUP BY MAKHOA
